 {{-- Script para inicializar el calendario --}}
    <script>
        document.addEventListener('DOMContentLoaded', function() {
            let formulario = document.querySelector("form");
            var calendarEl = document.getElementById('calendar');
            if (calendarEl) {
                var calendar = new FullCalendar.Calendar(calendarEl, {
                    locale: 'es',
                    headerToolbar: {
                        left: 'prev,next today',
                        center: 'title',
                        right: 'dayGridMonth, timeGridWeek,listWeek' // Vista inicial 
                    },
                    dateClick: function(info) {
                        $("#claseModal").modal("show");
                    }
                });
                calendar.render();
                document.getElementById("btnGuardar").addEventListener("click", function() {
                    const form = document.getElementById('eventoForm');
                    const datos = {
                        title: form.title.value,
                        descripcion: form.descripcion.value,
                        start: form.start.value,
                        end: form.end.value,
                    };

                    axios.post("{{ route('eventos.store') }}", datos, {
                            headers: {
                                'X-CSRF-TOKEN': document.querySelector('meta[name="csrf-token"]')
                                    .getAttribute('content')
                            }
                        })
                        .then((respuesta) => {
                            console.log(respuesta.data);
                            $("#claseModal").modal("hide"); // Close the modal after saving
                            window.location.reload(); // Refresh the page to see the new event
                        })
                        .catch(error => {
                            if (error.response) {
                                console.log(error.response.data);
                            }
                        });

                });
            }
        });
    </script>
